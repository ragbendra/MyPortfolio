Sample DAX Measure


YoY Growth = 
VAR CurrentYearSales = CALCULATE(
    SUM(Sales[Amount]),
    DATEADD(Calendar[Date], -1, YEAR)
)
RETURN
    DIVIDE(
        SUM(Sales[Amount]) - CurrentYearSales,
        CurrentYearSales,
        0
    )
